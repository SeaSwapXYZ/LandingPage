import { SeaSwapSDK } from '@seaswap/sdk';

const sdk = new SeaSwapSDK({ network: 'kaspa-mainnet' });

async function getQuote() {
  const quote = await sdk.getQuote({
    fromToken: 'KAS',
    toToken: 'USDC',
    amount: '100000000', // in smallest units
  });
  console.log(quote.toTokenAmount);
}

getQuote();
